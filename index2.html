def code_str(text_in="", step=2): result = "" for i in range(len(text_in)): if i % step == 0: result += text_in[i].upper() else: result += text_in[i] return result # Przykłady użycia print(code_str("Top", 2)) # Wynik: "ToP" print(code_str("secret", 2)) # Wynik: "SeCrEt" print(code_str("message", 2)) # Wynik: "MeSsAgE" def code_txt(*args): """ Funkcja code_txt() przyjmuje dowolną ilość argumentów i zwraca połączony łańcuch znaków. Parametry: *args - dowolna ilość argumentów tekstowych """ code_list = [] for arg in args: code_list.append(code_str(arg, step=2)) return ",".join(code_list) # Przykłady użycia print(code_txt("Top", "secret", "message", "for you")) # Wynik: "ToP,SeCrEt,MeSsAgE,FoR YoU"
